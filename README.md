# Crossword_CSP

Solving crosswords as a constraint satisfaction problem. 5 crossword of different sizes are included.

Example:

['#', '#', '#', '#', '#', '#']
['#', '_', '_', '_', '_', '#']
['#', '_', '_', '_', '#', '#']
['#', '_', '_', '_', '_', '#']
['#', '#', '#', '#', '#', '#']

['need', 'boat', 'ban', 'art', 'ore', 'ate']

Result:

['#', '#', '#', '#', '#', '#']
['#', 'b', 'o', 'a', 't', '#']
['#', 'a', 'r', 't', '#', '#']
['#', 'n', 'e', 'e', 'd', '#']
['#', '#', '#', '#', '#', '#']
